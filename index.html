function calcRec(){

  const qi=0;

  const opts=D[prod];

  let totalGap=0,totalOpts=0,cheapCount=0,expCount=0,midCount=0;

  const rows=[];

  opts.forEach(o=>{

    const us=o.op[qi];

    const cpx=CK.slice(1).map(k=>o[k]?o[k][qi]:null).filter(v=>v&&v>0);

    if(!cpx.length){rows.push({k:o.k,us,mn:null,avg:null,mx:null,pct:null,rec:0});return;}

    const mn=Math.min(...cpx),mx=Math.max(...cpx),avg=cpx.reduce((a,b)=>a+b,0)/cpx.length;

    const pct=us/avg;

    let rec=0;

    if(pct<0.80){rec=Math.round((avg*0.88-us)/us*100);cheapCount++;}

    else if(pct<0.90){rec=Math.round((avg*0.93-us)/us*100);cheapCount++;}

    else if(pct<0.98){rec=Math.round((avg*0.97-us)/us*100);midCount++;}

    else if(pct<1.05){rec=0;midCount++;}

    else{rec=Math.round((avg*1.0-us)/us*100);expCount++;}

    totalGap+=pct;totalOpts++;

    rows.push({k:o.k,us,mn,avg:Math.round(avg),mx,pct,rec:Math.max(-30,Math.min(50,rec))});

  });

  const avgPct=totalOpts>0?totalGap/totalOpts:1;

  const overallRec=avgPct<0.85?Math.round((0.90-avgPct)*100):avgPct<0.93?Math.round((0.95-avgPct)*100):avgPct<1.02?0:Math.round((1.0-avgPct)*100);

  return{rows,overallRec:Math.max(-20,Math.min(40,overallRec)),avgPct,cheapCount,midCount,expCount,totalOpts};

}
